    scene.py:

            functions:
                addComponent(component)
                drawComponents()
                onKeyPress(keys, mods)
                onMouseClick(mousePos)

            functions:
                initComponents()
                onKeyPress(keys, mods)
                onMouseClick(mousePos)
        class GameScene(Scene):

                drawBlock(block, position)
                drawPlayer()
                onKeyPress(keys, mods)
